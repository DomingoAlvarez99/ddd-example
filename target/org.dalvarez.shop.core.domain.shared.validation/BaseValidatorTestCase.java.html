package org.dalvarez.shop.core.domain.shared.validation;

import org.dalvarez.shop.core.shared.TestConfig;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import static org.junit.jupiter.api.Assertions.assertEquals;
import static org.junit.jupiter.api.Assertions.assertThrows;

@TestConfig
public class BaseValidatorTestCase<T> {

    private static final Logger logger = LoggerFactory.getLogger(BaseValidatorTestCase.class);

    protected final FieldValidator fieldValidator;

    public BaseValidatorTestCase(FieldValidator fieldValidator) {
        this.fieldValidator = fieldValidator;
    }

    protected void processValidationPassed(T value) throws ValidationNotPassedException {
        fieldValidator.validate("?", value);
    }

    protected void processValidationNotPassed(T value) {
        processValidationNotPassed(value, fieldValidator, fieldValidator.getCause());
    }

    protected void processValidationNotPassed(T value, FieldValidator fieldValidator) {
        processValidationNotPassed(value, fieldValidator, fieldValidator.getCause());
    }

    private void processValidationNotPassed(T value, FieldValidator fieldValidator, String cause) {
        String fieldName = "?";

        ValidationNotPassedException exception = assertThrows(ValidationNotPassedException.class, () -> fieldValidator.validate(fieldName, value));

        logger.info("Exception message: {}", exception.getMessage());
        String expectedMessage = String.format(ValidationNotPassedException.FORMAT, fieldName, value, fieldValidator.getClass().getSimpleName(), cause);

        logger.info("Expected message: {}", expectedMessage);
        assertEquals(expectedMessage, exception.getMessage());
    }

}
