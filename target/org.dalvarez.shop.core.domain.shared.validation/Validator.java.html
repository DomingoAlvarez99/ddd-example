package org.dalvarez.shop.core.domain.shared.validation;

import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

import java.lang.reflect.Field;
import java.util.ArrayList;
import java.util.List;
import java.util.Map;

public class Validator<T> {

    private static final Logger log = LoggerFactory.getLogger(Validator.class.getName());

    private final Map<String, FieldValidator> fieldValidators;

    protected Validator(Map<String, FieldValidator> fieldValidators) {
        this.fieldValidators = fieldValidators;
    }

    protected void validate(T instance) throws InvalidObjectException {
        List<String> validationErrors = new ArrayList<>();

        fieldValidators.forEach((fieldName, validator) -> {
            try {
                Field currentField = instance.getClass().getDeclaredField(fieldName);
                currentField.setAccessible(true);
                Object fieldValue = currentField.get(instance);

                validator.validate(fieldName, fieldValue);
            } catch (NoSuchFieldException e) {
                log.error("Field {} not found", fieldName);
                throw new IllegalArgumentException(e);

            } catch (IllegalAccessException e) {
                log.error("Could not access to the field {}", fieldName);
                throw new IllegalStateException(e);

            } catch (ValidationNotPassedException e) {
                validationErrors.add(e.getMessage());

            } catch (Exception e) {
                validationErrors.add("Unknown error {" + e.getMessage() + "}");
            }
        });

        if (!validationErrors.isEmpty())
            throw new InvalidObjectException(instance.getClass(), validationErrors);
    }

}
