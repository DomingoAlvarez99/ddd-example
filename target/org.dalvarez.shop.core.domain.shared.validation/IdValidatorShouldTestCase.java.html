package org.dalvarez.shop.core.domain.shared.validation;

import org.junit.jupiter.api.Test;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

class IdValidatorShouldTestCase extends BaseValidatorTestCase<Long> {

    private static final Logger logger = LoggerFactory.getLogger(IdValidatorShouldTestCase.class.getName());


    private IdValidatorShouldTestCase() {
        super(new IdValidator());
    }

    @Test
    void shouldPassTheValidation() throws ValidationNotPassedException {
        Long id = 1000L;

        processValidationPassed(id);
    }

    @Test
    void shouldNotPassTheValidation() {
        Long id = 0L;

        processValidationNotPassed(id);
    }

    @Test
    void shouldNotPassTheValidationCauseIsNull() {
        processValidationNotPassed(null, new NotNullValidator());
    }
    
}
