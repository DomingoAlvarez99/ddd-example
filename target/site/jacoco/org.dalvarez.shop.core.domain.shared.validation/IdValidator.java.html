package org.dalvarez.shop.core.domain.shared.validation;

public final class IdValidator extends InRangeValidator {

    static final Double MAX_VALUE = Double.MAX_VALUE;

    static final Double MIN_VALUE = 1D;

    private static final Integer CAUSE_DECIMALS = 2;

    public IdValidator() {
        super(MIN_VALUE, MAX_VALUE, CAUSE_DECIMALS);
    }

    @Override
    public void validate(String fieldName, Object fieldValue) throws ValidationNotPassedException {
        validate(fieldName, fieldValue, this.getClass());
    }


    public static void main(String[] args) {
        IdValidator idValidator = new IdValidator();
        System.out.println(idValidator.getCause());
    }

}
