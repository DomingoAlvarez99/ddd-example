package org.dalvarez.shop.core.domain.shared.validation;

public class GenericNotEmptyValidator extends NotEmptyValidator {

    private static final String CAUSE_MESSAGE = "value is empty";

    private static final FieldValidator notNullValidator = new NotNullValidator();

    @Override
    public void validate(String fieldName, Object fieldValue) throws ValidationNotPassedException {
        notNullValidator.validate(fieldName, fieldValue);

        if (fieldValue.toString().isBlank()) {
            throw new ValidationNotPassedException(fieldName, fieldValue, CAUSE_MESSAGE, this.getClass());
        }
    }

    @Override
    public String getCause() {
        return CAUSE_MESSAGE;
    }

}
