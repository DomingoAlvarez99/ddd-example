package org.dalvarez.shop.core.domain.shared.validation;

import java.util.List;

public class InRangeValidator implements FieldValidator {

    private final Number minValue;

    private final Number maxValue;

    private final Integer causeDecimals;

    private static final List<FieldValidator> validators = List.of(
            new NotNullValidator(),
            new GenericNotEmptyValidator()
    );

    public InRangeValidator(Number minValue, Number maxValue) {
        this.minValue = minValue;
        this.maxValue = maxValue;
        causeDecimals = 2;
    }

    public InRangeValidator(Number minValue, Number maxValue, Integer causeDecimals) {
        this.minValue = minValue;
        this.maxValue = maxValue;
        this.causeDecimals = causeDecimals;
    }

    @Override
    public void validate(String fieldName, Object fieldValue) throws ValidationNotPassedException {
        validate(fieldName, fieldValue, this.getClass());
    }

    protected void validate(String fieldName, Object fieldValue, Class<? extends FieldValidator> clazz) throws ValidationNotPassedException {
        for (FieldValidator validator : validators) {
            validator.validate(fieldName, fieldValue);
        }

        double valueParsed = Double.parseDouble(fieldValue.toString());

        if (valueParsed < minValue.doubleValue() || valueParsed > maxValue.doubleValue()) {
            throw new ValidationNotPassedException(fieldName, fieldValue, getCause(), clazz);
        }
    }

    @Override
    public String getCause() {
        String format = "the value is out of range [<minValue=%." + causeDecimals + "f>, <maxValue=%." + causeDecimals + "f>]";

        return String.format(
                format,
                minValue.doubleValue(),
                maxValue.doubleValue()
        );
    }

}
