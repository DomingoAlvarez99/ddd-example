package org.dalvarez.shop.core.domain.shared.validation;

public final class ValidationNotPassedException extends Exception {

    static final String FORMAT = "Field <%s=%s> has not passed the validator <%s> cause %s";

    public ValidationNotPassedException(String fieldName, Object fieldValue, String cause, Class<?> fieldValidator) {
        super(String.format(FORMAT, fieldName, fieldValue, fieldValidator.getSimpleName(), cause));
    }

}
