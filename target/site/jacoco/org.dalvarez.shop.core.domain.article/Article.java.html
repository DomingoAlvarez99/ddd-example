package org.dalvarez.shop.core.domain.article;

import org.dalvarez.shop.core.domain.shared.validation.FieldValidator;
import org.dalvarez.shop.core.domain.shared.validation.GenericNotEmptyValidator;
import org.dalvarez.shop.core.domain.shared.validation.IdValidator;
import org.dalvarez.shop.core.domain.shared.validation.InRangeValidator;
import org.dalvarez.shop.core.domain.shared.validation.InvalidObjectException;
import org.dalvarez.shop.core.domain.shared.validation.UuidValidator;
import org.dalvarez.shop.core.domain.shared.validation.Validator;

import java.util.Map;
import java.util.Objects;

public final class Article extends Validator<Article> {

    private static final Integer MIN_STOCK = 1;

    private static final Integer MAX_STOCK = 100;

    private static final Double MIN_PRICE = 0d;

    private static final Double MAX_PRICE = 1000d;

    private final Long id;

    private final String uuid;

    private final Integer stock;

    private final Double price;

    private final String name;

    private final String description;

    private static final Map<String, FieldValidator> fieldValidators = Map.of(
            "id", new IdValidator(),
            "uuid", new UuidValidator(),
            "stock", new InRangeValidator(MIN_STOCK, MAX_STOCK),
            "price", new InRangeValidator(MIN_PRICE, MAX_PRICE),
            "name", new GenericNotEmptyValidator(),
            "description", new GenericNotEmptyValidator()
    );

    private Article(Long id, String uuid, Integer stock, Double price, String name, String description) throws InvalidObjectException {
        super(fieldValidators);

        this.id = id;
        this.uuid = uuid;
        this.stock = stock;
        this.price = price;
        this.name = name;
        this.description = description;

        validate(this);
    }

    public static Article create(Long id, String uuid, Integer stock, Double price, String name, String description) throws InvalidObjectException {
        return new Article(id, uuid, stock, price, name, description);
    }

    public Long getId() {
        return id;
    }

    public String getUuid() {
        return uuid;
    }

    public Integer getStock() {
        return stock;
    }

    public Double getPrice() {
        return price;
    }

    public String getName() {
        return name;
    }

    public String getDescription() {
        return description;
    }

    @Override
    public boolean equals(Object o) {
        if (this == o)
            return true;

        if (o == null || getClass() != o.getClass())
            return false;

        Article article = (Article) o;

        return Objects.equals(id, article.id);
    }

    @Override
    public int hashCode() {
        return Objects.hash(id);
    }
}
